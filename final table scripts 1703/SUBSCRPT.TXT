     CREATE TABLE                                                       
       BMGVZP.SUBSCRIPTION                                              
        (                                                               
        SUBSCRIPTION_OID   DECIMAL(10) NOT NULL                         
         GENERATED BY DEFAULT                                           
         AS IDENTITY                                                    
         (                                                              
           START WITH 1199169816                                        
           INCREMENT BY 1                                               
           MINVALUE 1199169816                                          
           MAXVALUE 9999999999                                          
           NO CYCLE                                                     
           CACHE 20                                                     
           NO ORDER                                                     
         )                                                              
       ,SERVICE_ID   CHAR(50) NOT NULL WITH DEFAULT                     
                                                           
       ,SERVICE_TYPE   CHAR(20) NOT NULL WITH DEFAULT                   
                                                           
       ,SERVICE_ORDER_NUM   CHAR(8)                                     
                                                           
       ,STATUS   CHAR(1) NOT NULL WITH DEFAULT 'A'                      
                                                           
       ,STATUS_DATE   DATE NOT NULL WITH DEFAULT                        
       ,CUSTOMER_CODE   CHAR(3)                                         
                                                           
       ,CUSTOMER_ACCT_NUM   CHAR(50)                                    
                                                           
       ,ACCOUNT_TYPE   CHAR(10)                                         
                                                           
       ,BACKEND_SYSTEM   CHAR(10)                                       
                                                           
       ,REGION   CHAR(10)                                               
                                                           
       ,ORIG_SYSTEM_ID   CHAR(2)                                        
                                                           
       ,KEY_EFFECTIVE_DATE   DATE                                       
       ,STATE   CHAR(2)                                                 
                                                           
       ,CHANNEL_CODE   CHAR(2)                                          
                                                           
       ,BILL_PERIOD   CHAR(2)                                           
                                                           
       ,LAST_BILL_DATE   DATE                                           
       ,NEXT_BILL_DATE   DATE                                           
       ,USER_ID   DECIMAL(10) NOT NULL WITH DEFAULT                     
       ,TIMESTAMP   TIMESTAMP NOT NULL WITH DEFAULT                     
       ,INTERNAL_LABEL   CHAR(100)                                      
                                                           
       ,EXTERNAL_LABEL   CHAR(100)                                      
                                                           
       ,DUNS_CODE   CHAR(9)                                             
                                                           
       ,NATIVE_LOB   CHAR(4)                                            
                                                           
       ,BACKEND_DB   CHAR(10) NOT NULL WITH DEFAULT                     
                                                           
       ,PAPER_SUPPRESS_IND   CHAR(1)                                    
                                                           
       ,PAPER_SUPP_USER   DECIMAL(10)                                   
       ,BUYER_CODE   CHAR(3) NOT NULL WITH DEFAULT                      
                                                           
       ,REMIT_ACCT_IND   CHAR(1) NOT NULL WITH DEFAULT                  
                                                           
       ,PAPER_CHARGE_CODE   CHAR(2) NOT NULL WITH DEFAULT               
                                                           
       ,PAPER_CALL_DETAIL   CHAR(2) NOT NULL WITH DEFAULT               
                                                           
       ,PAPER_OIR_STATUS   CHAR(2) NOT NULL WITH DEFAULT                
                                                           
       ,ENTITY_CD   CHAR(5)                                             
                                                           
       ,ACCNT_NAME   CHAR(100)                                          
                                                           
       ,PAP_CONTRACT_AMEND   CHAR(1)                                    
                                                           
       ,ENTP_ID   DECIMAL(10)                                           
       ,PAPER_SUPP_ELIG   CHAR(1)                                       
                                                           
       ,SPLIT_BILL_IND   CHAR(1)                                        
                                                           
       ,SENSITIVITY_LVL   CHAR(2) NOT NULL WITH DEFAULT                 
                                                           
       ,ROOT_ACCOUNT   CHAR(1) NOT NULL WITH DEFAULT 'N'                
                                                           
       ,CREDIT_CARD_IND   CHAR(1) NOT NULL WITH DEFAULT                 
                                                           
       ,MIXED_SVC_IND   CHAR(1) NOT NULL WITH DEFAULT                   
                                                           
       ,SV_BILLED_IND   CHAR(1) NOT NULL WITH DEFAULT                   
                                                           
       ,OPCO   CHAR(5) NOT NULL WITH DEFAULT                            
                                                           
       ,REVLOC_ID   CHAR(3)                                             
                                                           
       ,DOMESTIC_INDC   CHAR(1)                                         
                                                           
       ,PREFERRED_CURRENCY   CHAR(3)                                    
                                                           
       ,Q2C_FLAG   CHAR(1)                                              
                                                           
       ,BILL_FMT   CHAR(3) NOT NULL WITH DEFAULT                        
                                                           
       ,PENDING_BILL_FMT   CHAR(3) NOT NULL WITH DEFAULT                
                                                           
       ,PREFERRED_LANG   CHAR(3)                                        
                                                           
       ,TAX_REG_NBR   CHAR(30)                                          
                                                           
       ,PREFER_CRC_OUTPUT   CHAR(1) NOT NULL WITH DEFAULT               
                                                           
       ,PAYMENT_TERMS   CHAR(25)                                        
                                                           
       ,ACCOUNT_CATEGORY   CHAR(25)                                     
                                                           
       ,ON_DEMAND_BILL_CYC   CHAR(1)                                    
                                                           
       ,GSAM_SENSITIVITY_LVL   CHAR(2) NOT NULL WITH DEFAULT            
                                                           
       ,AR_SYSTEM   CHAR(30) NOT NULL WITH DEFAULT                      
                                                           
       ,CONF_IND   CHAR(1)                                              
                                                           
       ,BILL_DAY   DECIMAL(2)                                           
       ,GCH_SENSITIVITY_LEVEL   CHAR(1)                                 
                                                           
        ,                                                               
         CONSTRAINT SUBSCRIPTION_OID                                    
         PRIMARY KEY                                                    
         (                                                              
          SUBSCRIPTION_OID                                              
         )                                                              
        )                                                               
     ;                                                                  
     CREATE UNIQUE                                                      
        INDEX BMGVZP.SBSCRIP1                                           
       ON BMGVZP.SUBSCRIPTION                                           
        (                                                               
         SUBSCRIPTION_OID ASC                                           
        )                                                    
     ;                                                                  
     CREATE UNIQUE                                                      
        INDEX BMGVZP.SBSCRIP2                                           
       ON BMGVZP.SUBSCRIPTION                                           
        (                                                               
         CUSTOMER_ACCT_NUM ASC                                          
        ,ORIG_SYSTEM_ID ASC                                             
        ,SUBSCRIPTION_OID ASC                                           
        )                                                     
     ;                                                                  
     CREATE UNIQUE                                                      
        INDEX BMGVZP.SBSCRIP3                                           
       ON BMGVZP.SUBSCRIPTION                                           
        (                                                               
         SERVICE_ORDER_NUM ASC                                          
        ,CUSTOMER_ACCT_NUM ASC                                          
        ,ORIG_SYSTEM_ID ASC                                             
        ,SUBSCRIPTION_OID ASC                                           
        )                                                   
     ;                                                                  
     CREATE UNIQUE                                                      
        INDEX BMGVZP.SBSCRIP4                                           
       ON BMGVZP.SUBSCRIPTION                                           
        (                                                               
         SERVICE_ID ASC                                                 
        ,SUBSCRIPTION_OID ASC                                           
        )                                                   
     ;                                                                  
     CREATE                                                             
        INDEX BMGVZP.SBSCRIP5                                           
       ON BMGVZP.SUBSCRIPTION                                           
        (                                                               
         ORIG_SYSTEM_ID ASC                                             
        ,SERVICE_TYPE ASC                                               
        ,SERVICE_ID ASC                                                 
        ,CUSTOMER_ACCT_NUM ASC                                          
        )                                                           
     ;
